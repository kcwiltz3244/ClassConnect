WHY THIS PATCH IS NEEDED
The selected name can appear in the search list but still fail during lookup when
the spreadsheet header is "Name" instead of "name", or when the public class list
does not include a row ID. This version normalizes the headers and searches by
the exact selected classmate name.

ADD TO THE doGet ACTION SWITCH:
case "lookupClassmate":
  result = lookupClassmate_(e.parameter.id, e.parameter.name);
  break;

ADD TO THE doPost ACTION SWITCH:
case "updateClassmate":
  result = updateClassmate_(payload);
  break;

ADD OR REPLACE THESE FUNCTIONS:

function ccNormalizeHeader_(value) {
  return String(value || "")
    .trim()
    .toLowerCase()
    .replace(/[^a-z0-9]/g, "");
}

function ccFindClassmatesSheet_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const preferredNames = ["Classmates", "Classmate", "Profiles", "Stories"];

  for (const name of preferredNames) {
    const sheet = ss.getSheetByName(name);
    if (sheet) return sheet;
  }

  // Fallback: find a sheet containing a recognizable name column.
  for (const sheet of ss.getSheets()) {
    if (sheet.getLastRow() < 1 || sheet.getLastColumn() < 1) continue;
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn())
      .getValues()[0]
      .map(ccNormalizeHeader_);
    if (
      headers.includes("name") ||
      headers.includes("fullname") ||
      headers.includes("classmatename")
    ) {
      return sheet;
    }
  }
  return null;
}

function ccObjectFromRow_(headers, rowValues, rowNumber) {
  const item = {};
  headers.forEach((header, index) => {
    item[header] = rowValues[index];
  });
  item.id = rowNumber;
  item.recordId = rowNumber;
  return item;
}

function lookupClassmate_(id, name) {
  const sh = ccFindClassmatesSheet_();
  if (!sh) return { item: null, error: "Classmates sheet not found." };

  const values = sh.getDataRange().getValues();
  if (values.length < 2) return { item: null };

  const rawHeaders = values[0].map(value => String(value || "").trim());
  const normalizedHeaders = rawHeaders.map(ccNormalizeHeader_);
  const requestedRow = Number(id);

  // Use the selected record ID when it is available and valid.
  if (requestedRow && requestedRow >= 2 && requestedRow <= sh.getLastRow()) {
    return {
      item: ccObjectFromRow_(rawHeaders, values[requestedRow - 1], requestedRow)
    };
  }

  // Find a likely classmate-name column regardless of capitalization or spacing.
  const nameAliases = [
    "name",
    "fullname",
    "classmatename",
    "studentname",
    "displayname"
  ];

  let nameColumn = -1;
  for (const alias of nameAliases) {
    nameColumn = normalizedHeaders.indexOf(alias);
    if (nameColumn !== -1) break;
  }

  if (nameColumn === -1) {
    return { item: null, error: "No name column was found." };
  }

  const wantedName = String(name || "")
    .trim()
    .toLowerCase()
    .replace(/\s+/g, " ");

  for (let rowIndex = 1; rowIndex < values.length; rowIndex++) {
    const sheetName = String(values[rowIndex][nameColumn] || "")
      .trim()
      .toLowerCase()
      .replace(/\s+/g, " ");

    if (sheetName === wantedName) {
      return {
        item: ccObjectFromRow_(rawHeaders, values[rowIndex], rowIndex + 1)
      };
    }
  }

  return { item: null };
}

function updateClassmate_(payload) {
  const sh = ccFindClassmatesSheet_();
  if (!sh) return { ok: false, error: "Classmates sheet not found." };

  const row = Number(payload.id || payload.recordId);
  if (!row || row < 2 || row > sh.getLastRow()) {
    return { ok: false, error: "Invalid classmate record." };
  }

  const rawHeaders = sh
    .getRange(1, 1, 1, sh.getLastColumn())
    .getValues()[0]
    .map(value => String(value || "").trim());

  const normalizedHeaders = rawHeaders.map(ccNormalizeHeader_);
  const rowValues = sh.getRange(row, 1, 1, rawHeaders.length).getValues()[0];

  const updates = {
    name: payload.name || "",
    fullname: payload.name || "",
    classmatename: payload.name || "",
    city: payload.city || "",
    state: payload.state || "",
    birthday: payload.birthday || "",
    career: payload.career || "",
    favoritememory: payload.favoriteMemory || "",
    email: payload.email || "",
    phone: payload.phone || "",
    bio: payload.bio || "",
    biography: payload.bio || "",
    showemail: payload.showEmail === true,
    showphone: payload.showPhone === true
  };

  normalizedHeaders.forEach((header, columnIndex) => {
    if (Object.prototype.hasOwnProperty.call(updates, header)) {
      rowValues[columnIndex] = updates[header];
    }
  });

  if (payload.profilePhotoData) {
    const photoAliases = ["profilephoto", "photo", "profileimage"];
    for (const alias of photoAliases) {
      const photoColumn = normalizedHeaders.indexOf(alias);
      if (photoColumn !== -1) {
        rowValues[photoColumn] = payload.profilePhotoData;
        break;
      }
    }
  }

  sh.getRange(row, 1, 1, rawHeaders.length).setValues([rowValues]);
  return { ok: true };
}
