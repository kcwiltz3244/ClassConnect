function lookupClassmate_(id,name){
  const sh=SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Classmates");
  if(!sh)return {item:null};

  const v=sh.getDataRange().getValues();
  if(v.length<2)return {item:null};

  const h=v[0].map(String);
  const requestedRow=Number(id);

  // Prefer the selected record ID when the public class list includes one.
  if(requestedRow && requestedRow>=2 && requestedRow<=sh.getLastRow()){
    const rowValues=v[requestedRow-1];
    const item={};
    h.forEach((x,j)=>item[x]=rowValues[j]);
    item.id=requestedRow;
    return {item};
  }

  // Fall back to the exact selected classmate name.
  const nc=h.indexOf("name");
  const wantedName=String(name||"").trim().toLowerCase();

  for(let i=1;i<v.length;i++){
    if(String(v[i][nc]||"").trim().toLowerCase()===wantedName){
      const item={};
      h.forEach((x,j)=>item[x]=v[i][j]);
      item.id=i+1;
      return {item};
    }
  }
  return {item:null};
}

function updateClassmate_(p){
  const sh=SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Classmates");
  if(!sh)return {ok:false};

  const row=Number(p.id);
  if(!row||row<2||row>sh.getLastRow())return {ok:false};

  const h=sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0].map(String);
  const r=sh.getRange(row,1,1,h.length).getValues()[0];
  const u={
    name:p.name||"",
    city:p.city||"",
    state:p.state||"",
    birthday:p.birthday||"",
    career:p.career||"",
    favoriteMemory:p.favoriteMemory||"",
    email:p.email||"",
    phone:p.phone||"",
    bio:p.bio||"",
    showEmail:p.showEmail===true,
    showPhone:p.showPhone===true
  };

  if(p.profilePhotoData)u.profilePhoto=p.profilePhotoData;
  h.forEach((x,j)=>{if(Object.prototype.hasOwnProperty.call(u,x))r[j]=u[x]});
  sh.getRange(row,1,1,h.length).setValues([r]);
  return {ok:true};
}

After saving the Apps Script, deploy a NEW VERSION of the web app and keep the same access settings used by ClassConnect.

PRIVACY NOTE:
Without email verification, anyone who can open ClassConnect and select a classmate's name may be able to open and change that story. Consider adding a private PIN later if stronger protection is desired.
